Quick Start

import com.NoticPro.Noticer;

public class MyApp {
  public static void main(String[] args) {
    Noticer.showNotic("Success!", "Task completed", Noticer.SUCCESS);
  }
}

Notification Types

Noticer.SUCCESS // Green checkmark, success sound
Noticer.WARNING // Red border, warning sound
Noticer.INFO // Info sound, auto-closes

Examples

// Enable sounds
Noticer.enableSounds();

// Show different notifications
Noticer.showNotic("Success", "File saved", Noticer.SUCCESS);
Noticer.showNotic("Warning", "Low disk space", Noticer.WARNING);
Noticer.showNotic("Info", "Update available", Noticer.INFO);

// Custom size notification
Noticer.showCustomSizeNotic(400, 150, "Custom", "Bigger notification", Noticer.INFO);

Sound Control

Noticer.enableSounds(); // Turn sounds ON
Noticer.disableSounds(); // Turn sounds OFF
boolean enabled = Noticer.areSoundsEnabled(); // Check status
