üéØ Complete Example
import com.flsl.fileio.File;
import com.flsl.utils.Arrays.Array;
import com.flsl.FGUI.fgui.*;

public class FLSLDemo {
  public static void main(String[] args) {
    // File operations
    File config = new File("config.txt");
    String settings = config.readAll();

    // Array operations
    Array numbers = new Array(new int[]{5, 2, 8, 1, 9});
    numbers.sort();
    System.out.println("Max: " + numbers.max());

    // GUI application
    FFrame app = new FFrame("FLSL Demo");
    FButton btn = new FButton("Process Data");
    app.add(btn);
    app.setSize(400, 300);
    app.setVisible(true);
  }
}
